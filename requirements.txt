# Python dependencies for Data Extraction & Regulation Mapping.
#
# Install everything:
#     pip install -r requirements.txt
#
# Versions are intentionally left unpinned; once you have a known-good
# environment, pin them with `pip freeze > requirements.lock.txt` for
# reproducible installs.
#
# Tkinter is required but ships with the standard python.org installer,
# so it is not listed here.

# --- PDF extraction & chunk review (Docling pipeline) ---
docling            # PDF -> document model + HybridChunker
docling-core       # PictureItem / doc types used by the extractor
pymupdf            # imported as `fitz`
pandas             # table dataframes
tqdm               # progress bars
colorama           # colored console logs

# --- EASA XML extraction & graph builder ---
xmltodict          # raw XML -> JSON
openpyxl           # Excel (.xlsx) output

# --- PDF -> Markdown ---
markitdown[pdf]    # MarkItDown with PDF support

# --- AI review (LLM calls to Blablador / DLR Ollama) ---
requests           # HTTP calls to the OpenAI-compatible endpoints
tiktoken           # token counting for the interaction log
# (colorama and pandas, listed above, are also used by the AI utilities)

# --- Evaluation metrics (column-analysis evaluation) ---
# Optional at runtime: without them the evaluator still computes the
# grounding check, Jaccard and a difflib similarity ratio.
nltk               # BLEU (Lexical_Overlap_Metrics)
rouge-score        # ROUGE-1/2/L (Lexical_Overlap_Metrics)
Levenshtein        # edit distance (Distance_w_Structural _Alignment)
jiwer              # word error rate (Distance_w_Structural _Alignment)
